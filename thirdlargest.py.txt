a=[10,20,50,1,60]
largest=float('-inf')
second_largest=float('-inf')
third_largest=float('-inf')
for i in range(len(a)):
    if a[i]>largest:
        third_largest=second_largest
        second_largest=largest
        largest=a[i]
    elif a[i]<largest and a[i]>second_largest:
        third_largest=second_largest
        second_largest=a[i]
    elif a[i]< largest and a[i]<second_largest and a[i]>third_largest:
        third_largest=a[i]
    
print(largest)
print(second_largest)
print(third_largest)
